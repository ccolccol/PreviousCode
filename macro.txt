Sub myInput_Calc()
    Dim des As String
    des = ActiveDocument.Name
    
    Dim fd As FileDialog
    Set fd = Application.FileDialog(msoFileDialogOpen)
    fd.InitialFileName = ActiveDocument.Path
    
    If fd.Show = -1 Then
        Dim sum(17), rowidx, colinitial, offset, sumidx, temp As Integer
        For i = 0 To 17
            sum(i) = 0
        Next
        
        Dim source As Variant
        For Each source In fd.SelectedItems
           Documents.Open (source)
           Windows(source).Activate
           
           For i = 0 To 13
                If i = 0 Then
                    offset = 0
                    rowidx = 3
                    colinitial = 2
                    sumidx = 14
                ElseIf i = 7 Then
                    offset = 0
                    rowidx = 5
                    colinitial = 2
                    sumidx = 15
                ElseIf i = 11 Then
                    offset = 0
                    rowidx = 7
                    colinitial = 2
                    sumidx = 16
                End If
                
                temp = Val(ActiveDocument.Tables(1).Cell(rowidx, colinitial + offset).Range)
                sum(i) = sum(i) + temp
                sum(sumidx) = sum(sumidx) + temp
                offset = offset + 1
           Next
           
           Windows(source).Close
        Next
    
        Windows(des).Activate
        For i = 0 To 13
            If i = 0 Then
                offset = 0
                rowidx = 3
                colinitial = 2
            ElseIf i = 7 Then
                offset = 0
                rowidx = 5
                colinitial = 2
            ElseIf i = 11 Then
                offset = 0
                rowidx = 7
                colinitial = 2
            End If
           
            ActiveDocument.Tables(1).Cell(rowidx, colinitial + offset).Range = sum(i)
            offset = offset + 1
            
            If i = 6 Then
                ActiveDocument.Tables(1).Cell(rowidx, colinitial + offset).Range = sum(14)
            ElseIf i = 10 Then
                ActiveDocument.Tables(1).Cell(rowidx, colinitial + offset).Range = sum(15)
            ElseIf i = 13 Then
                ActiveDocument.Tables(1).Cell(rowidx, colinitial + offset).Range = sum(16)
            End If
        Next
    End If
End Sub
